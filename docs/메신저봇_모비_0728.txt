/**
 * @name    심층 알람 (관리자/다중채팅방 적용)
 * @author  사용자
 * @version 7.2
 * @description 변수 중복 선언 오류를 수정한 버전입니다.
 */

const scriptName = "심층알람_v7.2";

// --- ▼ 설정 ▼ ---
const DEBUG_MODE = true; 
// [중요] 여기에 관리자의 userHash 값을 입력하세요. '$내해시' 명령어로 확인할 수 있습니다.
const ADMIN_USER_HASH = "여기에_관리자_해시값_입력"; 

// AppData에 channelId 목록을 저장하기 위한 키(Key)
const KALIX_IDS_KEY = "kalix_channel_ids";
const DAYAN_IDS_KEY = "dayan_channel_ids";
const TEST_ID_KEY = "test_channel_id"; // 테스트방은 하나만 지정
// --- ▲ 설정 ▲ ---

const debugLog = (message) => {
    if (DEBUG_MODE) console.log(`[${scriptName}] ${message}`);
};

const bot = BotManager.getCurrentBot();

/**
 * [API2] 카카오톡 메시지 수신 이벤트 리스너
 */
bot.addListener(Event.MESSAGE, (chat) => {
    try {
        if (!chat || !chat.room || !chat.author) return;

        const { room, content, author, channelId } = chat;
        const trimmedMsg = content.trim();
        const userHash = author.userHash;

        // [추가] 메시지를 보낸 유저의 이름과 해시를 저장합니다.
        // userHash가 유효한 경우에만 저장
        if (author.name && userHash) {
            // [수정] 변수 이름 변경 (userKey -> userKeyToSave)
            const userKeyToSave = "userhash_" + author.name;
            AppData.putString(userKeyToSave, userHash);
        }

        // --- 일반 사용자용 명령어 ---
        if (trimmedMsg === "$내해시") {
            chat.reply(`당신의 고유 해시값은 아래와 같습니다.\n\n${userHash}`);
            return;
        }
        
        // [추가] $유저정보 명령어 처리
        if (trimmedMsg.startsWith("$유저정보 ")) {
            const nameToLookup = trimmedMsg.substring(5).trim();
            if (!nameToLookup) {
                chat.reply("[오류] 조회할 유저의 이름을 입력해주세요.\n(예: $유저정보 강미루/말티즈)");
                return;
            }

            // [수정] 변수 이름 변경 (userKey -> userKeyToLookup)
            const userKeyToLookup = "userhash_" + nameToLookup;
            const foundHash = AppData.getString(userKeyToLookup);

            if (foundHash) {
                chat.reply(`[유저정보]\n${nameToLookup} 님의 id는 ${foundHash} 입니다.`);
            } else {
                chat.reply(`[오류] '${nameToLookup}' 님의 정보를 찾을 수 없습니다. 해당 유저가 최근에 채팅을 친 기록이 있는지 확인해주세요.`);
            }
            return;
        }
        
        if (trimmedMsg === "$진단") {
            const kalixIds = JSON.parse(AppData.getString(KALIX_IDS_KEY) || "[]");
            const dayanIds = JSON.parse(AppData.getString(DAYAN_IDS_KEY) || "[]");
            const debugInfo = `[메신저봇 진단]\n\n▼ 봇 인식 정보 ▼\nroom: '${room.name}'\nsender: '${author.name}'\nisGroupChat: ${room.isGroupChat}\nchannelId: ${channelId}\n\n▼ 등록된 알람방 ▼\n칼릭스: ${kalixIds.length}개\n데이안: ${dayanIds.length}개`;
            chat.reply(debugInfo);
            return;
        }

        // --- 관리자 전용 명령어 ---
        if (userHash === ADMIN_USER_HASH) {
            handleAdminCommands(chat);
        }

        // --- 테스트 명령어 (지정된 테스트방에서만) ---
        const testChannelId = AppData.getString(TEST_ID_KEY);
        if (testChannelId && String(channelId) === testChannelId && trimmedMsg.startsWith("$테스트 ")) {
            debugLog(`테스트 명령어 감지: ${content}`);
            handleTestCommand(content, chat);
        }

    } catch (e) {
        console.error(`Event.MESSAGE 처리 오류: ${e.name}: ${e.message}\n${e.stack}`);
    }
});

/**
 * 관리자 전용 설정 명령어를 처리하는 함수
 */
const handleAdminCommands = (chat) => {
    const { room, content, channelId } = chat;
    const trimmedMsg = content.trim();

    if (!channelId) {
        chat.reply("[오류] 이 채팅방의 고유 ID를 읽어올 수 없습니다.");
        return;
    }
    const channelIdStr = String(channelId);

    // --- 알람방 '추가' 명령어 ---
    if (trimmedMsg.startsWith("$알람설정 ")) {
        const server = trimmedMsg.split(" ")[1];
        const KEY = (server === "칼릭스") ? KALIX_IDS_KEY : (server === "데이안") ? DAYAN_IDS_KEY : null;
        if (!KEY) return;

        const ids = JSON.parse(AppData.getString(KEY) || "[]");
        if (ids.includes(channelIdStr)) {
            chat.reply(`[알림] 이 채팅방은 이미 '${server}' 알람방으로 등록되어 있습니다.`);
        } else {
            ids.push(channelIdStr);
            AppData.putString(KEY, JSON.stringify(ids));
            chat.reply(`[설정 완료] 이 채팅방이 '${server}' 알람방으로 추가되었습니다. (현재 ${ids.length}개)`);
        }
    }
    // --- 알람방 '삭제' 명령어 ---
    else if (trimmedMsg.startsWith("$알람삭제 ")) {
        const server = trimmedMsg.split(" ")[1];
        const KEY = (server === "칼릭스") ? KALIX_IDS_KEY : (server === "데이안") ? DAYAN_IDS_KEY : null;
        if (!KEY) return;

        let ids = JSON.parse(AppData.getString(KEY) || "[]");
        if (ids.includes(channelIdStr)) {
            ids = ids.filter(id => id !== channelIdStr);
            AppData.putString(KEY, JSON.stringify(ids));
            chat.reply(`[삭제 완료] 이 채팅방을 '${server}' 알람 목록에서 제거했습니다. (현재 ${ids.length}개)`);
        } else {
            chat.reply(`[알림] 이 채팅방은 '${server}' 알람방으로 등록되어 있지 않습니다.`);
        }
    }
    // --- 테스트방 '설정' 명령어 ---
    else if (trimmedMsg === "$테스트방설정") {
        AppData.putString(TEST_ID_KEY, channelIdStr);
        chat.reply(`[설정 완료] 이 채팅방이 테스트방으로 지정되었습니다.`);
    }
};


/**
 * [API2] 기기 알림 수신 이벤트 리스너
 */
bot.addListener(Event.NOTIFICATION_POSTED, (sbn, sm) => {
    try {
        if (sbn.getPackageName() === "com.android.chrome") {
            const extras = sbn.getNotification().extras;
            const notiTitle = extras.getString("android.title");
            
            if (notiTitle === "심층 검은 구멍 출현 알림") {
                const notiText = extras.getString("android.text");
                if (notiText) {
                    debugLog(`정상 알림 감지: ${notiText}`);
                    processAbyssNotification(notiText);
                }
            }
        }
    } catch (e) {
        console.error(`Event.NOTIFICATION_POSTED 처리 오류: ${e}`);
    }
});

const handleTestCommand = (msg, chat) => {
    const command = msg.substring(5).trim();
    let mockNotiText = "";

    if (command === "칼릭스") {
        mockNotiText = "칼릭스 서버의 얼음 협곡 지역 심층 구멍이 1개 발견되었습니다!";
        chat.reply("칼릭스 서버 알림을 테스트합니다.");
    } else if (command === "데이안") {
        mockNotiText = "데이안 서버의 불타는 사막 지역 심층 구멍이 1개 발견되었습니다!";
        chat.reply("데이안 서버 알림을 테스트합니다.");
    } else {
        chat.reply("잘못된 테스트 명령어입니다.");
        return;
    }
    processAbyssNotification(mockNotiText);
};

const processAbyssNotification = (notiText) => {
    try {
        const postTime = new Date();
        const hours = String(postTime.getHours()).padStart(2, '0');
        const minutes = String(postTime.getMinutes()).padStart(2, '0');
        const discoveryTime = `${hours}:${minutes}`;

        const messageBody = notiText.replace(/.*?서버의\s/, "");
        const finalMessage = `[심층 구멍 알람]\n${messageBody}\n발견시간: ${discoveryTime}`;
        
        let targetIdsKey = null;
        let targetRoomNameForLog = ""; 

        if (notiText.includes("칼릭스 서버")) {
            targetIdsKey = KALIX_IDS_KEY;
            targetRoomNameForLog = "칼릭스";
        } else if (notiText.includes("데이안 서버")) {
            targetIdsKey = DAYAN_IDS_KEY;
            targetRoomNameForLog = "데이안";
        }

        if (targetIdsKey) {
            const targetChannelIds = JSON.parse(AppData.getString(targetIdsKey) || "[]");
            if (targetChannelIds.length > 0) {
                debugLog(`'${targetRoomNameForLog}' 알람방 ${targetChannelIds.length}개에 메시지 전송 시도...`);
                targetChannelIds.forEach(id => {
                    bot.send(id, finalMessage);
                });
            } else {
                console.error(`'${targetRoomNameForLog}' 알람방이 등록되어 있지 않습니다.`);
            }
        } else {
            console.error("알림 내용에서 서버를 찾을 수 없습니다.");
        }
    } catch (e) {
        console.error(`processAbyssNotification 오류: ${e}`);
    }
};

debugLog(`${scriptName} 스크립트가 로드되었습니다.`);
